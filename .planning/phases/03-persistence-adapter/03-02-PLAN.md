using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.DependencyInjection;
using Storage.Infrastructure.Persistence.SqlServer;
using Storage.Infrastructure.Persistence.SqlServer.Extensions;
using Storage.Infrastructure.Persistence.SqlServer.Interceptors;
using Testcontainers.MsSql;

namespace Storage.Infrastructure.Persistence.SqlServer.Tests.Infrastructure;

public sealed class SqlServerFixture : IAsyncLifetime
{
    private readonly MsSqlContainer _container = new MsSqlBuilder()
        .WithImage("mcr.microsoft.com/mssql/server:2022-latest")
        .Build();

    public string ConnectionString => _container.GetConnectionString();

    public async Task InitializeAsync()
    {
        await _container.StartAsync();
        // Apply migrations once per fixture lifetime
        await using var ctx = CreateDbContext();
        await ctx.Database.MigrateAsync();
    }

    public async Task DisposeAsync()
        => await _container.DisposeAsync();

    public StorageDbContext CreateDbContext()
    {
        var interceptor = new SoftDeleteInterceptor();
        var options = new DbContextOptionsBuilder<StorageDbContext>()
            .UseSqlServer(ConnectionString, sql => sql.EnableRetryOnFailure(3))
            .AddInterceptors(interceptor)
            .Options;
        return new StorageDbContext(options);
    }

    public IServiceScope CreateScope()
    {
        var services = new ServiceCollection();
        services.AddSqlServerPersistence(ConnectionString);
        var provider = services.BuildServiceProvider();
        return provider.CreateScope();
    }
}

using FluentAssertions;
using Microsoft.EntityFrameworkCore;
using Storage.Infrastructure.Persistence.SqlServer.Seeders;
using Storage.Infrastructure.Persistence.SqlServer.Tests.Infrastructure;

namespace Storage.Infrastructure.Persistence.SqlServer.Tests;

[Collection("SqlServer")]
public class MigrationTests : IClassFixture<SqlServerFixture>
{
    private readonly SqlServerFixture _fixture;
    public MigrationTests(SqlServerFixture fixture) => _fixture = fixture;

    [Fact]
    public async Task SchemaCreatedCleanly()
    {
        await using var db = _fixture.CreateDbContext();

        // Verify all 6 tables exist
        var tables = await db.Database
            .SqlQueryRaw<string>("SELECT name FROM sys.tables WHERE type = 'U'")
            .ToListAsync();

        tables.Should().Contain(["Files", "FileCategories", "FileVersions",
            "FilePermissions", "FileTags", "AuditLog"]);

        // Verify check constraints
        var constraints = await db.Database
            .SqlQueryRaw<string>("SELECT name FROM sys.check_constraints")
            .ToListAsync();

        constraints.Should().Contain("CK_Files_Status");
        constraints.Should().Contain("CK_FilePermissions_PrincipalType");

        // Verify at least one non-clustered index exists on Files
        var indexes = await db.Database
            .SqlQueryRaw<string>(
                "SELECT i.name FROM sys.indexes i " +
                "JOIN sys.tables t ON i.object_id = t.object_id " +
                "WHERE t.name = 'Files' AND i.type = 2") // type 2 = NONCLUSTERED
            .ToListAsync();

        indexes.Should().NotBeEmpty("at least one non-clustered index should exist on Files");
    }

    [Fact]
    public async Task SeedIsIdempotent()
    {
        // Use a dedicated scope to avoid cross-test contamination
        await using var db1 = _fixture.CreateDbContext();

        // Clear existing seed data to test from clean state
        db1.FileCategories.RemoveRange(db1.FileCategories);
        await db1.SaveChangesAsync();

        var seeder1 = new FileCategorySeeder(db1);
        await seeder1.SeedAsync();
        var countAfterFirstRun = await db1.FileCategories.CountAsync();

        // Second run on same context
        var seeder2 = new FileCategorySeeder(db1);
        await seeder2.SeedAsync();
        var countAfterSecondRun = await db1.FileCategories.CountAsync();

        countAfterFirstRun.Should().Be(7, "seeder must insert exactly 7 categories");
        countAfterSecondRun.Should().Be(7, "second run must be a no-op");
    }
}

        var files = Enumerable.Range(0, 5)
            .Select(i => CreateTestFile(tenantId))
            .ToList();

using DomainFile = Storage.Domain.Entities.File;
using FluentAssertions;
using Microsoft.EntityFrameworkCore;
using Storage.Domain.Enums;
using Storage.Infrastructure.Persistence.SqlServer.Tests.Infrastructure;

namespace Storage.Infrastructure.Persistence.SqlServer.Tests;

[Collection("SqlServer")]
public class SoftDeleteTests : IClassFixture<SqlServerFixture>

    private readonly SqlServerFixture _fixture;
    public SoftDeleteTests(SqlServerFixture fixture) => _fixture = fixture;

    [Fact]
    public async Task SoftDeletedFilesExcluded()
    {
        var tenantId = Guid.NewGuid();
        await using var db = _fixture.CreateDbContext();

        var file = DomainFile.Create(tenantId, "svc", "image", "photo.jpg", "image/jpeg", 512);
        await db.Files.AddAsync(file);
        await db.SaveChangesAsync();

        // Soft-delete: set Status to Deleted — interceptor stamps DeletedAt
        file.SetStatus(FileStatus.Deleted);
        await db.SaveChangesAsync();

        // Default query (respects query filter) must not return the file
        await using var db2 = _fixture.CreateDbContext();
        var found = await db2.Files.FirstOrDefaultAsync(f => f.Id == file.Id);
        found.Should().BeNull("soft-deleted file must be invisible to default queries");

    [Fact]
    public async Task HardDelete_FindsSoftDeleted()
    {
        var tenantId = Guid.NewGuid();
        await using var db = _fixture.CreateDbContext();

        var file = DomainFile.Create(tenantId, "svc", "image", "remove.jpg", "image/jpeg", 256);
        await db.Files.AddAsync(file);
        await db.SaveChangesAsync();

        // Soft-delete
        file.SetStatus(FileStatus.Deleted);
        await db.SaveChangesAsync();

        // HardDelete must succeed even though file is soft-deleted
        await using var db2 = _fixture.CreateDbContext();
        var repo = new Storage.Infrastructure.Persistence.SqlServer.Repositories.FileRepository(db2);
        var result = await repo.HardDeleteAsync(file.Id, tenantId, CancellationToken.None);

        result.Should().BeTrue("hard delete must find and remove soft-deleted row");

        // Verify physical deletion
        await using var db3 = _fixture.CreateDbContext();
        var gone = await db3.Files.IgnoreQueryFilters()
            .FirstOrDefaultAsync(f => f.Id == file.Id);
        gone.Should().BeNull("file must be physically removed after HardDeleteAsync");
    }
}
